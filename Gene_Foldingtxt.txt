\section{Gene Folding}

\begin{frame}{INTRODUCTION:}

\begin{itemize}
  \item \textbf{To obtain the shortest genetic sequence from a given input genetic sequence using genetically optimized organic folding (GOOF) process.}
  \item \textbf{Our main aim is to find the length of the shortest sequence or length of the string.}
\end{itemize}

\end{frame}

\begin{frame}{APPROACH:}

\begin{itemize}
\item \textbf{This code takes a string input and returns the length of the shortest string.}
\item \textbf{The function 'GOOF' implements this logic by using two while loops that check the first and second half of the string respectively.}
\item \textbf{Finally it returns the length of the shortened string.}
\end{itemize}
\end{frame}

\begin{frame}{LEARNINGS:}
\begin{itemize}
    \item \textbf{We learned the process of the GOOF.}

    \item \textbf{We learned how to give input by using command line argument.}
\end{itemize}
